**Goal:** Build `@ollive/llm-sdk` — a standalone, storage-agnostic SDK that wraps any `LLMProvider`, captures inference metadata (latency, TTFT, token usage, status, previews), redacts PII at completion time, and ships validated `InferenceLog` payloads to a configurable ingestion URL over a buffered, non-blocking HTTP transport that never adds latency to the user-facing streamed response. The SDK reuses the on-the-wire contract (`inferenceLogSchema`) from `@ollive/shared` — it does **not** redefine it — and does **not** depend on `@ollive/db`.

**Architecture:** The SDK is a pure instrumentation layer (PRD §13). Its public surface is `withLogging(provider, config)`, a decorator that returns a new `LLMProvider` with the identical interface (SDK1). The provider abstraction (`LLMProvider`/`ChatRequest`/`StreamChunk`/`CallContext`, PRD §14) lets application code depend only on our interface, never a vendor SDK (PA1). The default adapter, `GoogleProvider`, is backed by the **Vercel AI SDK** (`ai` + `@ai-sdk/google`) and **normalizes** provider token-usage field names into our `{ promptTokens, completionTokens, totalTokens }` shape — normalizing provider differences is precisely the adapter's job (PA2). A `Redactor` scrubs PII from the fully-assembled text **then** truncation runs (redact-then-truncate, SDK10), so a PII token straddling the preview cut can never leak a fragment; redaction is fail-closed (SDK13). The transport buffers logs in a bounded in-memory queue, flushes on size/interval, retries with exponential backoff + jitter, and drops on overflow/exhaustion with a local warning (SDK4/SDK5). The whole logging path runs at completion, off the streaming hot path — chunks are yielded to the caller immediately (SDK10, NFR1).

**Tech Stack:** TypeScript 5.7, pnpm workspaces, Vitest 3 (root workspace runner), Zod 3, `@ollive/shared` (workspace), Vercel AI SDK `ai` ^5.0.0 + `@ai-sdk/google` ^2.0.0, Node 20 built-ins (`node:crypto` for `randomUUID`, `globalThis.fetch`, `node:http` for the test mock server).

**Context:** The repo root is the existing git repository at `chatbot-ollive/` (already contains `docs/PRD.md`, Plan 1's `package.json`, `pnpm-workspace.yaml`, `tsconfig.base.json`, `vitest.workspace.ts`, and the `@ollive/shared` + `@ollive/db` packages). ```ts
import type { Redactor } from '../types';

/** All PII type keys this redactor can emit, so counts always include every type. */
const PII_TYPES = ['email', 'phone', 'credit_card', 'ssn', 'ip', 'iban', 'api_key'] as const;
type PiiType = (typeof PII_TYPES)[number];

const PLACEHOLDER: Record<PiiType, string> = {
  email: '[EMAIL]',
  phone: '[PHONE]',
  credit_card: '[CREDIT_CARD]',
  ssn: '[SSN]',
  ip: '[IP]',
  iban: '[IBAN]',
  api_key: '[API_KEY]',
};

/** Luhn check for credit-card validation (PRD §13 SDK11). */
function luhnValid(digits: string): boolean {
  const d = digits.replace(/\D/g, '');
  if (d.length < 13 || d.length > 19) return false;
  let sum = 0;
  let alt = false;
  for (let i = d.length - 1; i >= 0; i--) {
    let n = d.charCodeAt(i) - 48;
    if (alt) {
      n *= 2;
      if (n > 9) n -= 9;
    }
    sum += n;
    alt = !alt;
  }
  return sum % 10 === 0;
}

/** Shannon entropy (bits per char) — used to flag high-entropy secret tokens. */
function shannonEntropy(s: string): number {
  const freq: Record<string, number> = {};
  for (const ch of s) freq[ch] = (freq[ch] ?? 0) + 1;
  let h = 0;
  for (const c of Object.values(freq)) {
    const p = c / s.length;
    h -= p * Math.log2(p);
  }
  return h;
}

/**
 * Pattern-based, deterministic, in-process PII redactor (PRD §13 SDK11/SDK12).
 * Order matters: structured patterns (card/ssn/iban/ip) run before the generic
 * high-entropy secret pass so a card number isn't mis-tagged as an API key.
 */
export class PatternRedactor implements Redactor {
  redact(text: string): { text: string; counts: Record<string, number> } {
    const counts: Record<string, number> = {};
    for (const t of PII_TYPES) counts[t] = 0;
    let out = text;

    const replace = (re: RegExp, type: PiiType, guard?: (m: string) => boolean): void => {
      out = out.replace(re, (m) => {
        if (guard && !guard(m)) return m;
        counts[type] += 1;
        return PLACEHOLDER[type];
      });
    };

    // Email — the whole address (incl. local part) is replaced
    replace(/[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}/g, 'email');
    // IBAN (2 letters + 2 digits + up to 30 alnum) — before generic key/entropy
    replace(/\b[A-Z]{2}\d{2}[A-Z0-9]{10,30}\b/g, 'iban');
    // Credit card: a digit then 12–18 more digits, optionally space/dash separated
    // (13–19 digits total); Luhn-guarded so non-card 16-digit runs are not flagged.
    replace(/\b\d(?:[ -]?\d){12,18}\b/g, 'credit_card', (m) => luhnValid(m));
    // SSN (US): 3-2-4
    replace(/\b\d{3}-\d{2}-\d{4}\b/g, 'ssn');
    // IPv4
    replace(/\b(?:(?:25[0-5]|2[0-4]\d|1?\d?\d)\.){3}(?:25[0-5]|2[0-4]\d|1?\d?\d)\b/g, 'ip');
    // Phone: a run starting with optional + and a digit, then digits/space/()/.-,
    // ending in a digit; guarded to 7–15 total digits. Runs AFTER card/ssn/ip so it
    // can't swallow those (they're already replaced with placeholders by now).
    replace(/\+?\d[\d ().-]{5,}\d/g, 'phone', (m) => {
      const digits = m.replace(/\D/g, '').length;
      return digits >= 7 && digits <= 15;
    });
    // Known secret-key prefixes (sk-, pk-, ghp_, AKIA, xoxb-, etc.)
    replace(/\b(?:sk|pk|rk)-[A-Za-z0-9]{16,}\b/g, 'api_key');
    replace(/\b(?:ghp|gho|ghu|ghs|ghr)_[A-Za-z0-9]{20,}\b/g, 'api_key');
    replace(/\bAKIA[0-9A-Z]{16}\b/g, 'api_key');
    replace(/\bxox[baprs]-[A-Za-z0-9-]{10,}\b/g, 'api_key');
    // High-entropy generic tokens (length >= 24, entropy > 3.5 bits/char, mixed alnum)
    replace(/\b[A-Za-z0-9_-]{24,}\b/g, 'api_key', (m) => {
      const hasUpper = /[A-Z]/.test(m);
      const hasLower = /[a-z]/.test(m);
      const hasDigit = /\d/.test(m);
      return hasUpper && hasLower && hasDigit && shannonEntropy(m) > 3.5;
    });

    return { text: out, counts };
  }
}

/** Pass-through redactor used when redaction is 'off' (PRD §13 SDK9). */
export class NoopRedactor implements Redactor {
  redact(text: string): { text: string; counts: Record<string, number> } {
    return { text, counts: {} };
  }
}

/**
 * Optional cheap-LLM redactor (PRD §13 SDK11). Opt-in, NOT enabled by default:
 * it costs latency + a model call, so the PRD recommends running it at the
 * ingestion backstop (IN9) rather than inline. Supplied as a real minimal shape
 * an implementer can wire a small/fast model into; until configured it composes
 * with the deterministic pattern pass so behavior never silently weakens.
 */
export class LlmRedactor implements Redactor {
  constructor(
    private readonly base: Redactor = new PatternRedactor(),
    private readonly classify?: (text: string) => { text: string; counts: Record<string, number> },
  ) {}

  redact(text: string): { text: string; counts: Record<string, number> } {
    const first = this.base.redact(text);
    if (!this.classify) return first;
    const second = this.classify(first.text);
    const counts = { ...first.counts };
    for (const [k, v] of Object.entries(second.counts)) counts[k] = (counts[k] ?? 0) + v;
    return { text: second.text, counts };
  }
}

/** Resolve a Redactor from config (PRD §13 SDK Interfaces). */
export function createRedactor(
  mode: 'off' | 'pattern' | 'llm',
  override?: Redactor,
): Redactor {
  if (override) return override;
  if (mode === 'off') return new NoopRedactor();
  // 'llm' without an override degrades to the deterministic pattern detector
  // (LLM mode is opt-in and not enabled by default — PRD §13 SDK11).
  return new PatternRedactor();
}
```

import type { LLMProvider } from './types';

/** A factory that constructs a provider instance (PRD §14 PA4). */
export type ProviderFactory = () => LLMProvider;

/**
 * Maps a provider name -> factory. Adding a provider = register one adapter
 * (FR14/PA3): no other code changes. Configured by env at app startup.
 */
export class ProviderRegistry {
  private readonly factories = new Map<string, ProviderFactory>();

  register(name: string, factory: ProviderFactory): this {
    this.factories.set(name, factory);
    return this;
  }

  has(name: string): boolean {
    return this.factories.has(name);
  }

  create(name: string): LLMProvider {
    const factory = this.factories.get(name);
    if (!factory) {
      throw new Error(`No LLM provider registered for "${name}"`);
    }
    return factory();
  }

  names(): string[] {
    return [...this.factories.keys()];
  }
}

/** Shape of the AI SDK's usage object across majors (some fields may be absent). */
interface SdkUsage {
  promptTokens?: number; // ai v4
  completionTokens?: number; // ai v4
  inputTokens?: number; // ai v5/v6
  outputTokens?: number; // ai v5/v6
  totalTokens?: number;
}

/**
 * Normalize the AI SDK usage object to our canonical
 * { promptTokens, completionTokens, totalTokens } (PRD §14). This is the
 * adapter's job: absorb provider/SDK-version differences here so the rest of
 * the system sees one shape.
 *
 * NOTE: pinned to ai@5.x => reads inputTokens/outputTokens. If you pin ai@4.x
 * (per Task 3 Step 1), change `u.inputTokens`->`u.promptTokens` and
 * `u.outputTokens`->`u.completionTokens`.
 */
function normalizeUsage(u: SdkUsage): { promptTokens: number; completionTokens: number; totalTokens: number } {
  const prompt = u.inputTokens ?? u.promptTokens ?? 0;
  const completion = u.outputTokens ?? u.completionTokens ?? 0;
  const total = u.totalTokens ?? prompt + completion;
  return { promptTokens: prompt, completionTokens: completion, totalTokens: total };
}

/** Normalize the AI SDK finishReason to our StreamChunk['finishReason'] set. */
function normalizeFinishReason(r: string | undefined): StreamChunk['finishReason'] {
  switch (r) {
    case 'stop':
      return 'stop';
    case 'length':
      return 'length';
    case 'content-filter':
    case 'content_filter':
      return 'content_filter';
    case 'error':
      return 'error';
    default:
      return 'stop';
  }
}

/** Gemini adapter backed by the Vercel AI SDK (PRD §14 PA2). name = 'google'. */
export class GoogleProvider implements LLMProvider {
  readonly name = 'google';

  async *streamChat(
    req: ChatRequest,
    opts?: { signal?: AbortSignal; context?: CallContext },
  ): AsyncIterable<StreamChunk> {
    const result = streamText({
      model: google(req.model),
      messages: req.messages,
      abortSignal: opts?.signal,
      temperature: req.temperature,
      maxOutputTokens: req.maxOutputTokens,
    });

    for await (const delta of result.textStream) {
      yield { delta };
    }

    const usage = normalizeUsage((await result.usage) as SdkUsage);
    const finishReason = normalizeFinishReason((await result.finishReason) as string | undefined);
    yield { usage, finishReason };
  }
}

/** Factory for the provider registry (PRD §14 PA4). */

  return new GoogleProvider();

import http from 'node:http';
import type { AddressInfo } from 'node:net';
import type { ChatRequest, LLMProvider, StreamChunk, CallContext } from '../src/types';

/**
 * Scripted fake provider. Yields the given chunks in order. Optionally throws
 * after N chunks (to test error status), or honors an AbortSignal (to test
 * cancelled status). Never calls a real model.
 */
export class FakeProvider implements LLMProvider {
  readonly name: string;
  constructor(
    private readonly chunks: StreamChunk[],
    private readonly opts: {
      name?: string;
      throwAfter?: number; // throw after yielding this many chunks
      throwError?: Error;
      delayMs?: number; // delay before the FIRST chunk (to make TTFT measurable)
      abortAfter?: number; // simulate honoring abort: stop + throw AbortError after N chunks if signalled
    } = {},
  ) {
    this.name = opts.name ?? 'fake';
  }

  async *streamChat(
    _req: ChatRequest,
    opts?: { signal?: AbortSignal; context?: CallContext },
  ): AsyncIterable<StreamChunk> {
    let i = 0;
    for (const chunk of this.chunks) {
      if (i === 0 && this.opts.delayMs) {
        await new Promise((r) => setTimeout(r, this.opts.delayMs));
      }
      if (opts?.signal?.aborted) {
        const err = new Error('aborted');
        err.name = 'AbortError';
        throw err;
      }
      yield chunk;
      i += 1;
      if (this.opts.throwAfter !== undefined && i >= this.opts.throwAfter) {
        throw this.opts.throwError ?? new Error('provider boom');
      }
      if (this.opts.abortAfter !== undefined && i >= this.opts.abortAfter && opts?.signal) {
        const err = new Error('aborted');
        err.name = 'AbortError';
        throw err;
      }
    }
  }
}

export interface ReceivedRequest {
  auth: string | undefined;
  body: unknown;
}

/**
 * Local mock ingestion HTTP server. Records every POSTed body + Authorization
 * header. `failTimes` makes the first N requests return 503 (to test retry).
 */
export async function makeMockIngestionServer(opts: { failTimes?: number } = {}) {
  const received: ReceivedRequest[] = [];
  let remainingFailures = opts.failTimes ?? 0;

  const server = http.createServer((req, res) => {
    const chunks: Buffer[] = [];
    req.on('data', (c) => chunks.push(c as Buffer));
    req.on('end', () => {
      const raw = Buffer.concat(chunks).toString('utf8');
      let body: unknown;
      try {
        body = JSON.parse(raw);
      } catch {
        body = raw;
      }
      received.push({ auth: req.headers.authorization, body });
      if (remainingFailures > 0) {
        remainingFailures -= 1;
        res.statusCode = 503;
        res.end(JSON.stringify({ error: 'unavailable' }));
        return;
      }
      res.statusCode = 202;
      res.end(JSON.stringify({ accepted: true }));
    });
  });

  await new Promise<void>((resolve) => server.listen(0, resolve));
  const { port } = server.address() as AddressInfo;

  return {
    url: `http://127.0.0.1:${port}/v1/logs`,
    received,
    async close() {
      await new Promise<void>((resolve) => server.close(() => resolve()));
    },
  };
}

/** Merge two redaction-count maps. */
function mergeCounts(
  a: Record<string, number>,
  b: Record<string, number>,
): Record<string, number> {
  const out: Record<string, number> = { ...a };
  for (const [k, v] of Object.entries(b)) out[k] = (out[k] ?? 0) + v;
  return out;
}

/** Build the assembled input text from the request messages (last user turn first). */
function assembleInput(req: ChatRequest): string {
  const lastUser = [...req.messages].reverse().find((m) => m.role === 'user');
  return lastUser?.content ?? req.messages.at(-1)?.content ?? '';
}

/**
 * Redact-then-truncate (PRD §13 SDK10). Returns the redacted+truncated text and
 * the redaction counts. Fail-closed (SDK13): if the redactor throws, returns
 * `{ text: undefined }` so the caller drops the preview rather than ship raw text.
 */
function redactThenTruncate(
  redactor: Redactor,
  raw: string,
  maxChars: number,
): { text?: string; counts: Record<string, number> } {
  try {
    const { text, counts } = redactor.redact(raw);
    return { text: text.slice(0, maxChars), counts };
  } catch {
    return { text: undefined, counts: {} };
  }
}

/**
 * Decorator (PRD §13 SDK1): wraps a provider, captures inference metadata,
 * redacts + truncates previews at completion, builds an InferenceLog, validates
 * it with the shared schema, and hands it to the sink. Adds zero latency to the
 * streamed response — chunks are yielded to the caller immediately; the log is
 * assembled and shipped only after the stream finishes/errors/aborts.
 *
 * `sink` is optional for production callers (defaults are wired by withLogging's
 * companion factory in Task 5); tests inject a fake sink.
 */
export function withLogging(
  provider: LLMProvider,
  config: InferenceLoggerConfig,
  sink: LogSink,
): LLMProvider {
  const previewMaxChars = config.previewMaxChars ?? PREVIEW_MAX_CHARS;
  const redactor = createRedactor(config.redaction ?? 'pattern', config.redactor);

  return {
    name: provider.name,
    async *streamChat(
      req: ChatRequest,
      opts?: { signal?: AbortSignal; context?: CallContext },
    ): AsyncIterable<StreamChunk> {
      const requestId = randomUUID();
      const startedAtMs = Date.now();
      const startedAt = new Date(startedAtMs).toISOString();
      let ttftMs: number | undefined;
      let outputText = '';
      let usage: StreamChunk['usage'];
      let finishReason: StreamChunk['finishReason'];
      let status: InferenceLog['status'] = 'success';
      let errorPayload: { code: string; message: string; providerCode?: string } | null = null;

      const ship = (): void => {
        const completedAtMs = Date.now();
        const completedAt = new Date(completedAtMs).toISOString();

        const inputRedacted = redactThenTruncate(redactor, assembleInput(req), previewMaxChars);
        const outputRedacted = redactThenTruncate(redactor, outputText, previewMaxChars);
        const redactions = mergeCounts(inputRedacted.counts, outputRedacted.counts);

        const preview: { input?: string; output?: string } = {};
        if (inputRedacted.text !== undefined) preview.input = inputRedacted.text;
        if (outputRedacted.text !== undefined) preview.output = outputRedacted.text;

        const callerMeta = opts?.context?.metadata ?? {};
        const context: InferenceLog['context'] = {};
        if (opts?.context?.conversationId) context.conversationId = opts.context.conversationId;
        if (opts?.context?.messageId) context.messageId = opts.context.messageId;
        if (opts?.context?.userId) context.userId = opts.context.userId;

        const candidate = {
          requestId,
          timestamp: completedAt,
          provider: provider.name,
          model: req.model,
          status,
          context,
          timing: {
            startedAt,
            completedAt,
            latencyMs: completedAtMs - startedAtMs,
            ...(ttftMs !== undefined ? { timeToFirstTokenMs: ttftMs } : {}),
          },
          usage: status === 'success' ? (usage ?? null) : null,
          preview,
          error: errorPayload,
          metadata: {
            ...callerMeta,
            stream: true,
            ...(req.temperature !== undefined ? { temperature: req.temperature } : {}),
            ...(req.maxOutputTokens !== undefined ? { maxOutputTokens: req.maxOutputTokens } : {}),
            ...(finishReason !== undefined ? { finishReason } : {}),
            redactions,
          },
        };

        // Validate against the shared contract; on validation failure, drop the
        // log locally (never throw on the logging path — NFR5).
        const parsed = inferenceLogSchema.safeParse(candidate);
        if (parsed.success) {
          sink.enqueue(parsed.data);
        } else {
          console.warn('[llm-sdk] dropping invalid inference log', parsed.error.flatten());
        }
      };

      try {
        for await (const chunk of provider.streamChat(req, opts)) {
          if (chunk.delta !== undefined) {
            if (ttftMs === undefined) ttftMs = Date.now() - startedAtMs;
            outputText += chunk.delta;
          }
          if (chunk.usage !== undefined) usage = chunk.usage;
          if (chunk.finishReason !== undefined) finishReason = chunk.finishReason;
          yield chunk; // forward to caller IMMEDIATELY — zero added latency
        }
      } catch (err) {
        const e = err as Error;
        if (e?.name === 'AbortError' || opts?.signal?.aborted) {
          status = 'cancelled';
          errorPayload = null;
        } else {
          status = 'error';
          errorPayload = { code: 'provider_error', message: e?.message ?? 'provider error' };
        }
        ship();
        throw err; // re-throw so the caller's stream surfaces the failure
      }
      ship();
    },
  };
}
```

```ts
import type { InferenceLog } from '@ollive/shared';
import type { LogSink } from '../logging/logger';

export interface TransportConfig {
  ingestionUrl: string;
  apiKey: string;
  maxBufferSize?: number; // default 500 (SDK7)
  flushIntervalMs?: number; // default 1000 (SDK7)
  maxRetries?: number; // default 3 (SDK7)
  baseBackoffMs?: number; // default 200; exponential base for retry backoff
  autoFlushOnFull?: boolean; // default true; flush when buffer reaches capacity
  fetchImpl?: typeof fetch; // injectable for tests; defaults to globalThis.fetch
}

const sleep = (ms: number): Promise<void> => new Promise((r) => setTimeout(r, ms));

/**
 * Buffered, non-blocking HTTP shipper (PRD §13 SDK4/SDK5/SDK7). Logs enqueue
 * into a bounded in-memory buffer; a background flush (size- or interval-
 * triggered) POSTs them to the ingestion URL with `Authorization: Bearer
 * <apiKey>`. Each ship retries with exponential backoff + jitter up to
 * `maxRetries`; on buffer overflow or exhausted retries the log is dropped with
 * a local warning. The caller never awaits delivery (enqueue returns instantly).
 */
export class BufferedHttpTransport implements LogSink {
  private readonly buffer: InferenceLog[] = [];
  private readonly maxBufferSize: number;
  private readonly flushIntervalMs: number;
  private readonly maxRetries: number;
  private readonly baseBackoffMs: number;
  private readonly autoFlushOnFull: boolean;
  private readonly doFetch: typeof fetch;
  private readonly timer: ReturnType<typeof setInterval>;
  private flushing = false;
  private closed = false;

  constructor(private readonly config: TransportConfig) {
    this.maxBufferSize = config.maxBufferSize ?? 500;
    this.flushIntervalMs = config.flushIntervalMs ?? 1000;
    this.maxRetries = config.maxRetries ?? 3;
    this.baseBackoffMs = config.baseBackoffMs ?? 200;
    this.autoFlushOnFull = config.autoFlushOnFull ?? true;
    this.doFetch = config.fetchImpl ?? globalThis.fetch.bind(globalThis);
    this.timer = setInterval(() => {
      void this.flush();
    }, this.flushIntervalMs);
    // Don't keep the process alive solely for the flush timer.
    (this.timer as unknown as { unref?: () => void }).unref?.();
  }

  bufferSize(): number {
    return this.buffer.length;
  }

  /** Non-blocking: drop-oldest on overflow; size-triggered flush is fire-and-forget. */
  enqueue(log: InferenceLog): void {
    if (this.closed) return;
    if (this.buffer.length >= this.maxBufferSize) {
      this.buffer.shift(); // drop oldest
      console.warn('[llm-sdk] transport buffer overflow — dropping oldest log');
    }
    this.buffer.push(log);
    if (this.autoFlushOnFull && this.buffer.length >= this.maxBufferSize) {
      void this.flush();
    }
  }

  /** Drains the buffer, shipping each entry. Never throws. */
  async flush(): Promise<void> {
    if (this.flushing) return;
    this.flushing = true;
    try {
      while (this.buffer.length > 0) {
        const log = this.buffer.shift()!;
        await this.shipWithRetry(log);
      }
    } finally {
      this.flushing = false;
    }
  }

  private async shipWithRetry(log: InferenceLog): Promise<void> {
    for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
      try {
        const res = await this.doFetch(this.config.ingestionUrl, {
          method: 'POST',
          headers: {
            'content-type': 'application/json',
            authorization: `Bearer ${this.config.apiKey}`,
          },
          body: JSON.stringify(log),
        });
        if (res.ok || res.status === 202) return; // delivered
        // 4xx (except 408/429) is non-retryable: a bad payload won't get better.
        if (res.status >= 400 && res.status < 500 && res.status !== 408 && res.status !== 429) {
          console.warn(`[llm-sdk] dropping log: non-retryable status ${res.status}`);
          return;
        }
        // else fall through to retry (5xx/408/429)
      } catch (err) {
        // network error — retry
        if (attempt === this.maxRetries) {
          console.warn('[llm-sdk] dropping log after network failures', (err as Error)?.message);
          return;
        }
      }
      if (attempt < this.maxRetries) {
        const backoff = this.baseBackoffMs * 2 ** attempt;
        const jitter = Math.random() * backoff;
        await sleep(backoff + jitter);
      } else {
        console.warn(`[llm-sdk] dropping log after ${this.maxRetries} retries`);
        return;
      }
    }
  }

  /** Stops the timer and flushes any remaining buffered logs. */
  async close(): Promise<void> {
    this.closed = true;
    clearInterval(this.timer);
    await this.flush();
  }
}
```
